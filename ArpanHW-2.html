import pandas as pd
from sktime.forecasting.model_selection import temporal_train_test_split
from sktime.forecasting.base import ForecastingHorizon
from sktime.forecasting.exp_smoothing import ExponentialSmoothing
from sktime.performance_metrics.forecasting import mean_absolute_percentage_error
import matplotlib.pyplot as plt

# Loading data
data = pd.read_csv('hw2_data.csv')
data['Date'] = pd.to_datetime(data['Date'])
data.set_index('Date', inplace=True)

# Missing values handle 
data = data.resample('b').interpolate('linear').ffill().bfill()

# Selecting SPY series and data split
spy_series = data['SPY']
y_train, y_test = temporal_train_test_split(spy_series, test_size=0.2)
fh = ForecastingHorizon(y_test.index, is_relative=False)

def apply_and_plot_smoothing_methods(y_train, y_test, fh):
    results = {}
    methods = {
        "SES": ExponentialSmoothing(sp=1, seasonal=None),
        "Holt_additive": ExponentialSmoothing(trend='add', sp=12, seasonal=None),
        "Holt_multiplicative": ExponentialSmoothing(trend='mul', sp=12, seasonal=None),
        "Holt-Winters_additive": ExponentialSmoothing(trend='add', sp=12, seasonal='additive'),
        "Holt-Winters_multiplicative": ExponentialSmoothing(trend='mul', sp=12, seasonal='multiplicative')
    }

    for name, model in methods.items():
        model.fit(y_train)
        y_pred = model.predict(fh)
        sMAPE = mean_absolute_percentage_error(y_test, y_pred)
        results[name] = (y_pred, sMAPE)

        plt.figure(figsize=(10, 4))
        plt.title(f"{name} - sMAPE: {sMAPE:.2f}")
        plt.plot(y_train.index, y_train, label='Train')
        plt.plot(y_test.index, y_test, label='Test', color='gray')
        plt.plot(y_pred.index, y_pred, label='Forecast', color='red')
        plt.xlabel('Year')  
        plt.ylabel('Stock Prices')  
        plt.legend()
        plt.show()

    return results

forecast_results = apply_and_plot_smoothing_methods(y_train, y_test, fh)
for method, (predictions, error) in forecast_results.items():
    print(f"{method} Forecasting Error (sMAPE): {error:.2f}")

Holt-Winters multiplicative model works best among those tested,has the lowest sMAPE at 0.13, indicates the most accurate forecasts among the models tested.These models might not be the best for to model stock price data due to the efficient market hypothesis, which suggests that stock prices reflect all available information and are thus inherently unpredictable. However, they could be useful for smooting out short-term fluctuations and indentifying broader terms. These models are better suited for forecasting data with consistent patterns and trends without too many external disruptions.
